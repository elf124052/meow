// POST to Apps Script

async function apiPost(body) {

  if (!SCRIPT_URL) throw new Error('尚未設定 Script URL');

  const res = await fetch(SCRIPT_URL, {

    method: 'POST',

    body: JSON.stringify(body),

  });

  const json = await res.json();

  if (!json.ok) throw new Error(json.error || 'API error');

  return json;

}


// GET all data from Sheets

async function fetchAll() {

  if (!SCRIPT_URL) return;

  showLoading('從 Google Sheets 讀取資料...');

  setSyncStatus('saving', '正在同步...');

  try {

    const res = await fetch(SCRIPT_URL + '?action=getAll');

    const json = await res.json();

    if (!json.ok) throw new Error(json.error);

    const d = json.data;


    // Rebuild state from flat arrays

    S.cats  = d.cats  || [];

    S.expos = d.expos || [];


    // Per-cat tables

    const catTables = ['health','vaccines','deworming','body','feeders','litter'];

    catTables.forEach(t => {

      S[t] = {};

      (d[t] || []).forEach(row => {

        const catId = row.catId;

        if (!catId) return;

        if (!S[t][catId]) S[t][catId] = [];

        S[t][catId].push(row);

      });

    });


    // Restore current cat if valid

    if (_currentCat && !S.cats.find(c=>c.id===_currentCat)) {

      _currentCat = S.cats[0]?.id || null;

    }

    if (!_currentCat && S.cats.length) _currentCat = S.cats[0].id;


    setSyncStatus('ok', '✅ 已同步 · ' + new Date().toLocaleTimeString('zh-TW'));

    renderCatTabs();

    renderPage(_page);

  } catch(e) {

    setSyncStatus('err', '❌ 同步失敗：' + e.message);

  } finally {

    hideLoading();

  }

}

  try {

    await apiPost({ action:'save', sheet:sheetName, data:arr });

    setSyncStatus('ok', '✅ 已儲存 · ' + new Date().toLocaleTimeString('zh-TW'));

  } catch(e) {

    setSyncStatus('err', '❌ 儲存失敗：' + e.message);

    throw e;

  }

}

function calcAge(bd){if(!bd)return'未知';const n=new Date(),d=new Date(bd);let y=n.getFullYear()-d.getFullYear(),m=n.getMonth()-d.getMonth();if(m<0){y--;m+=12;}return y>0?`${y} 歲 ${m} 個月`:`${m} 個月`;}

function daysUntil(date){const d=new Date(date),n=new Date();n.setHours(0,0,0,0);d.setHours(0,0,0,0);return Math.ceil((d-n)/86400000);}

function openInGCal(ev){const d=ev.date.replace(/-/g,'');const nd=new Date(ev.date);nd.setDate(nd.getDate()+1);const de=nd.toISOString().slice(0,10).replace(/-/g,'');window.open(`https://calendar.google.com/calendar/render?action=TEMPLATE&text=${encodeURIComponent(ev.title)}&dates=${d}/${de}&details=${encodeURIComponent(ev.note||'')}`, '_blank');}

async function addRecord(sheetName, catKey, obj) {

  if(!cid()) return alert('請先選擇貓咪');

  if(!SCRIPT_URL) return alert('請先設定 Google Sheets 連線！');

  ensArr(S[sheetName], catKey || cid());

  S[sheetName][catKey || cid()].push(obj);

  renderPage(_page);

  await appendRow(sheetName, obj);

}


async function delRecord(sheetName, id) {

  if(!cid()) return;

  if(!SCRIPT_URL) return alert('請先設定 Google Sheets 連線！');

  if(!confirm('確定刪除？')) return;

  // Remove from local state

  if(S[sheetName][cid()]) S[sheetName][cid()] = S[sheetName][cid()].filter(r=>r.id!==id);

  renderPage(_page);

  // Build full flat array for this sheet and save

  const flat = Object.values(S[sheetName]).flat();

  await saveSheet(sheetName, flat);

}


async function delExpo(id) {

  S.expos = S.expos.filter(e=>e.id!==id);

  renderExpo();

}
